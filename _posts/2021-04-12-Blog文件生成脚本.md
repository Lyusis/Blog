```shell
###
 # @Description: 生成Blog使用的MD文件
 # @version: 0.1
 # @Author: Lyu Zhongqin
 # @Date: 2021-04-12 12:17:55
 # @LastEditors: Lyu Zhongqin
 # @LastEditTime: 2021-04-12 18:27:13
### 

#!/bin/bash

# BEGIN
echo "生成blog文件开始"
time=$(date "+%Y-%m-%d")
# 目标文件夹地址
description="./Documents/GitHub/Lyusis_Blog/_posts"
# 文件名
file=$description/$time-$1.md

# 生成MD文件
touch $description/$time-$1.md

# 检测文件内容
cat $description/$time-$1.md | while read line; 
    do
        if [ -z $line ]; then 
            break;
        else
            echo "已存在文件, 请确认"
            exit 1
        fi
    done

if [ $? -eq 1 ]; then
    exit 1
fi

# header
echo "---" >> $file
echo "layout: mypost" >> $file
echo "title: $1" >> $file
echo "categories: [$2]" >> $file
echo "---" >> $file

# END
if [ $? -eq 0 ]; then
    echo "blog文件生成完成"
else 
    echo "blog文件生成失败"
fi
```
